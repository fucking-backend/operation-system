```bash
# 读取参数
# 第9个变量之后，需要在变量数字周围加花括号
total=$[ $1*$2 ]
echo bash name is $0
echo first paramer is $1
echo second paramer is $2
echo total value is $total


# 读取脚本名
bashName=$(basename $0)
# 中括号左右两边和等号左右两边都需要加空格
if [ $bashName = "addNum" ] 
then	
	total=$[ $1+$2 ]
elif [ $bashName = "mulNum" ]
then
	total=$[ $1*$2 ]
fi
echo total = $total


# 测试参数，-n判断是否存在
if [ -n "$1" ]
then 
	echo Hello $1
else
	echo sorry，请输入参数
fi


# 参数统计，$# 包含脚本运行时参数个数
echo 有 $# 个参数
if [ $# -ne 2 ]
then
	echo $1*$2 = $[ $1*$2 ]
fi
echo 最后一个参数 ${!#}


# $*和$@,都表示所有变量
count=1
for param in "$*" # $*将所有变量当做单个参数
do 
	echo "\$* Parameter #$count=$param"
	count=$[ $count+1 ]
done

count=1
for param in "$@" # $@ 单独处理每个变量
do 
	echo "\$@ Parameter #$count=$param"
	count=$[ $count+1 ]
done


# shift移动变量，可以指定一次移动多个位置
count=1
while [ -n "$1" ]
do 
	echo "Parameter #$count=$1"
    # 等号这里不能有空格
	count=$[ $count + 2 ]
	shift 2
done
```
