```bash
# 读取参数
# 第9个变量之后，需要在变量数字周围加花括号
total=$[ $1*$2 ]
echo bash name is $0
echo first paramer is $1
echo second paramer is $2
echo total value is $total


# 读取脚本名
bashName=$(basename $0)
# 中括号左右两边和等号左右两边都需要加空格
if [ $bashName = "addNum" ] 
then	
	total=$[ $1+$2 ]
elif [ $bashName = "mulNum" ]
then
	total=$[ $1*$2 ]
fi
echo total = $total


# 测试参数，-n判断是否存在
if [ -n "$1" ]
then 
	echo Hello $1
else
	echo sorry，请输入参数
fi


# 参数统计，$# 包含脚本运行时参数个数
echo 有 $# 个参数
if [ $# -ne 2 ]
then
	echo $1*$2 = $[ $1*$2 ]
fi
echo 最后一个参数 ${!#}


# $*和$@,都表示所有变量
count=1
for param in "$*" # $*将所有变量当做单个参数
do 
	echo "\$* Parameter #$count=$param"
	count=$[ $count+1 ]
done

count=1
for param in "$@" # $@ 单独处理每个变量
do 
	echo "\$@ Parameter #$count=$param"
	count=$[ $count+1 ]
done


# shift移动变量，可以指定一次移动多个位置
count=1
while [ -n "$1" ]
do 
	echo "Parameter #$count=$1"
    # 等号这里不能有空格
	count=$[ $count + 2 ]
	shift 2
done


# 处理简单选项
while [ -n "$1" ]
do 
	case "$1" in
		-a) echo "Found the -a option";;
		-b) echo "Found the -b option";;
		-c) echo "Found the -c option";;
		*)  echo "$1 is not an option";;
	esac
	shift  # 移动变量
done


# 分离参数和选项
while [ -n "$1" ]
do 
	case "$1" in
		-a) echo "Found the -a option";;
		-b) echo "Found the -b option";;
		-c) echo "Found the -c option";;
		--) shift
		    break;;	
		*)  echo "$1 is not an option";;
	esac
	shift
done

count=1
for param in "$@"
do 
	echo "Parameter #$count: $param"
	count=$[ $count+1 ]
done


# 处理带值的选项
while [ -n "$1" ]
do 
	case "$1" in
		-a) echo "Found the -a option";;
		-b) param="$2"  # 要求-b携带额外参数
		    echo "Found the -b option, with parameter value $param"
		    shift;;
		-c) echo "Found the -c option";;
		--) shift
		    break;;	
		*)  echo "$1 is not an option";;
	esac
	shift
done

count=1
for param in "$@"
do 
	echo "Parameter #$count: $param"
	count=$[ $count+1 ]
done



# getopt命令处理【合并选项】
# 需要参数的选项后加一个冒号
# -q 忽略不在optionstring中的选项
set -- $(getopt -q ab:cd "$@")

while [ -n "$1" ]
do 
	case "$1" in
		-a) echo "Found the -a option";;
		-b) param="$2"
		    echo "Found the -b option, with parameter value $param"
		    shift;;
		-c) echo "Found the -c option";;
		--) shift
		    break;;	
		*)  echo "$1 is not an option";;
	esac
	shift
done

count=1
for param in "$@"
do 
	echo "Parameter #$count: $param"
	count=$[ $count+1 ]
done


# getopts选项
# 去掉错误optstring之前加一个冒号
# 要求有参数的选项后加一个冒号
# OPTIND记录选项位置
while getopts :ab:cd opt
do 
	case "$opt" in
	a) echo "Found the -a option";;
	b) echo "Found the -b option, with value $OPTARG";;
	c) echo "Found the -c option";;
	d) echo "Found the -d option";;
	*) echo "Unknow option $opt"
	esac
done

shift $[ $OPTIND - 1 ]

count=1
for param in "$@"
do
	echo "Parameter $count: $param"
	count=$[ $count + 1 ]
done


# 选项标准化



# 获得用户输入。基本读取（read），超时处理(-t)，隐藏方式读取(-s)，从文件中读取

# -n1 告诉read命令接受单个字符后退出
read -n1 -p "continue [Y/N]?" answer
case $answer in
Y|y) echo "继续操练";;
N|n) echo "休息五分钟";;
esac

# 从文件中读取
count=1
cat /etc/passwd | while read line
do
	echo "Line $count:$line"
	count=$[ $count + 1 ]
done
echo "完结撒花"

```
