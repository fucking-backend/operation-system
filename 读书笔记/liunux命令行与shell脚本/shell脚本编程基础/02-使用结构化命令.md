## 使用结构化命令

```bash
# if-then语句 【只能测试命令退出状态码】
if pwd # 判断退出状态码
then
	echo "It is if-then"
	echo "It worked"
fi

if IamNotCommand
then
	echo "It worked"
fi
echo "outside"


# if-then-else语句
if IamNotCommand
then
	echo "It worked"
else
	echo "command not found"
fi
echo "outside"

# 嵌套if和elif


# test 命令
v1="full"
if test $v1 # 确定变量是否有内容
then
	echo "the $v1 return True"
else
	echo "the $v1 return False"
fi

v2=""
if test $v2
then
	echo "the $v2 return True"
else
	echo "the $v2 return False"
fi


# 方括号定义测试条件。三种
# ！！！！！注意：方括号与内容前后需要有空格
# 数值、字符串、文件
vn=10
vs="echo饭饭"
vf=~/Desktop/fubao-learning
if [ $vn -ge 9 ] # 数值
then 
	echo "$vn >= 9"
else 
	echo "$vn < 9"
fi


if [ -n $vs ] # 字符串
then 
	echo "$vs 长度非0"
else 
	echo "$vs 长度0"
fi

if [ -d $vf ] # 文件
then 
	echo "$vf 存在并且是目录"
else 
	echo "$vf 不存在"
fi

# 符合条件测试，两种
# && 和 ||

# if-then高级特性。两种
# 用于数学表达式的双括号
v1=10
if (($v1 ** 2 > 90))
then 
	((v2 = $v1 ** 2))
	echo "The square of $v1 is $v2"
fi
# 用于高级字符串处理的的双括号
if [[ $USER == u* ]]
then 
	echo "HELLO $USER"
else 
	echo "sorry $USER"
fi



# case命令
case $USER in
root | ubuntu)
	echo "Welcome $USER";;
mino)
	echo "user mino";;
*)
	echo "defaul";;
esac
```
